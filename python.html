Match/Caso (Match/Case)


# Ejemplo básico con match/case
def http_error(status):
    match status:
        case 400:
            return "Bad request"
        case 401 | 403 | 404:
            return "Not allowed"
        case 418:
            return "I'm a teapot"
        case _:
            return "Something else"

print(http_error(404)) # Salida: Not allowed

Módulo 'random'


import random

# Número flotante aleatorio entre 0.0 y 1.0
print(random.random()) # Ej: 0.12345

# Número entero aleatorio en un rango (excluye el límite superior)
print(random.randrange(0, 10)) # Ej: 0, 1, ..., 9

# Número entero aleatorio en un rango (incluye ambos límites)
print(random.randint(0, 10)) # Ej: 0, 1, ..., 10

# Seleccionar un elemento aleatorio de una secuencia
lista = [1, 2, 3, 4, 5]
print(random.choice(lista)) # Ej: 3

# Reordenar una secuencia aleatoriamente (shuffle)
random.shuffle(lista)
print(lista) # Ej: [2, 5, 1, 4, 3]

# Seleccionar múltiples elementos únicos de una secuencia
print(random.sample(lista, 2)) # Ej: [1, 4]

# Abrir y cerrar un archivo de texto (modo lectura 'r')

    f.write("Hola, Python!\n")
    f.write("Este es un archivo de prueba.")

Modos de Apertura


# Modos básicos
"r"  -> Leer (predeterminado). El archivo debe existir.
"w"  -> Escribir. Crea el archivo si no existe o lo trunca si ya existe.
"a"  -> Anexar. Crea el archivo si no existe o añade contenido al final.
"x"  -> Creación exclusiva. Falla si el archivo ya existe.

# Combinados con lectura y escritura
"r+" -> Leer y escribir sin truncar.
"w+" -> Escribir y leer, truncando el archivo al abrir.
"a+" -> Anexar y leer, manteniendo el contenido previo.

# Sufijos
"b"  -> Modo binario (ej: "rb" para leer binario).
"t"  -> Modo texto (predeterminado).

Lectura de Archivos


# Leer todo el contenido
with open("mi_archivo.txt", "r") as f:
    contenido = f.read()
    print(contenido)

# Leer línea por línea
with open("mi_archivo.txt", "r") as f:
    for linea in f:
        print(linea.strip()) # .strip() elimina saltos de línea y espacios en blanco

# Leer todas las líneas en una lista
with open("mi_archivo.txt", "r") as f:
    lineas = f.readlines()
    print(lineas)

Escritura de Archivos


# Escribir una cadena
with open("nuevo_archivo.txt", "w") as f:
    f.write("Primera línea.\n")
    f.write("Segunda línea.\n")

# Escribir una lista de cadenas
lista_de_lineas = ["Línea 1\n", "Línea 2\n", "Línea 3\n"]
with open("otro_archivo.txt", "w") as f:
    f.writelines(lista_de_lineas)

Indexación y Slicing

# Slicing de filas

Manejo de Excepciones

Estructura Básica try-except


try:
    resultado = 10 / 0
except ZeroDivisionError:
    print("¡Error: No se puede dividir por cero!")

Múltiples Bloques except


try:
    valor = int("abc")
    lista = [1, 2]
    print(lista[3])
except ValueError:
    print("¡Error: No se pudo convertir a entero!")
except IndexError:
    print("¡Error: Índice fuera de rango!")
except Exception as e:
    print(f"Ocurrió un error inesperado: {e}")

Bloques else y finally


try:
    num1 = 10
    num2 = 2
    resultado = num1 / num2
except ZeroDivisionError:
    print("Error: División por cero.")
else:
    print(f"El resultado es: {resultado}")
finally:
    print("La ejecución ha finalizado (siempre se ejecuta)."

Lanzar Excepciones Personalizadas (raise)


def verificar_edad(edad):
    if edad < 0:
        raise ValueError("La edad no puede ser negativa.")
    print(f"Edad válida: {edad}")

try:
    verificar_edad(-5)
except ValueError as e:
    print(f"Error: {e}")

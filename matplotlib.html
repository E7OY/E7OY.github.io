Tipos de Gráficas Comunes

Gráfica de Líneas


x = np.linspace(0, 10, 100)
y = np.sin(x)
plt.plot(x, y)
plt.title("Gráfica de Seno")
plt.xlabel("Eje X")
plt.ylabel("Eje Y")
plt.grid(True)
plt.show()

Gráfica de Dispersión (Scatter Plot)


x = np.random.rand(50)
y = np.random.rand(50)
plt.scatter(x, y)
plt.title("Gráfica de Dispersión")
plt.xlabel("Eje X")
plt.ylabel("Eje Y")
plt.show()

Histograma


data = np.random.randn(1000)
plt.hist(data, bins=30)
plt.title("Histograma")
plt.xlabel("Valor")
plt.ylabel("Frecuencia")
plt.show()

Gráfica de Barras


categorias = ['A', 'B', 'C', 'D']
valores = [23, 56, 12, 49]
plt.bar(categorias, valores)
plt.title("Gráfica de Barras")
plt.xlabel("Categoría")
plt.ylabel("Valor")
plt.show()

Gráfica de Pastel (Pie Chart)


tamanos = [15, 30, 45, 10]
etiquetas = ['Manzanas', 'Naranjas', 'Bananas', 'Uvas']
plt.pie(tamanos, labels=etiquetas, autopct='%1.1f%%', startangle=90)
plt.title("Gráfica de Pastel")
plt.axis('equal') # Asegura que el pastel sea circular
plt.show()

Múltiples Gráficas y Subplots

Subplots Básicos


fig, axes = plt.subplots(1, 2, figsize=(10, 4))
axes[0].plot([1, 2, 3])
axes[0].set_title("Primer Subplot")
axes[1].scatter([1, 2, 3], [3, 2, 1])
axes[1].set_title("Segundo Subplot")
plt.tight_layout()
plt.show()

Subplots con Gridspec


import matplotlib.gridspec as gridspec

fig = plt.figure(figsize=(10, 8))
gs = gridspec.GridSpec(2, 2)

ax1 = fig.add_subplot(gs[0, :])
ax1.plot(np.random.rand(10))
ax1.set_title("Gráfica Superior")

ax2 = fig.add_subplot(gs[1, 0])
ax2.hist(np.random.randn(100))
ax2.set_title("Histograma Inferior Izquierdo")

ax3 = fig.add_subplot(gs[1, 1])
ax3.scatter(np.random.rand(20), np.random.rand(20))
ax3.set_title("Dispersión Inferior Derecho")

plt.tight_layout()
plt.show()
